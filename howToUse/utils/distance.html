# 计算两个经纬度之间的距离

计算两个经纬度之间的距离

# 示例

const distanceRes = hnsdk.distance([40,40] , [50,50] , {
  units : 'kilometers'
})

console.log(distanceRes)

# 参数

入参类型描述
from[number,number]起始点
to[number,number]目标点
optionsObject可配置项

# options

选项类型描述
unitsString单位，可选的有 degrees、radians、miles、kilometers

# 返回

计算后的数字结果
